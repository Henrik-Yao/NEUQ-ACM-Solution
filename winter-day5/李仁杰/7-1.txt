#include <iostream>
#include <algorithm>
using namespace std;

int main(){
    int n, *a, ans = 0;
    cin >> n;
    a = new int[n];
    for(int i = 0; i < n; i++){
        cin >> a[i];
    }
    sort(a, a+n);
    for(int i = 1; i < n; i++){
        if(a[i] - a[i - 1] == 1) ans ++;
    }
    cout << ans;
}
