## 7-1 生化危机

```c++
#include <iostream>
using namespace std;

typedef struct 
{
	int arcs[1000][1000];
	int v,a; 
}graph;

int main()
{
	int n;
	int now,d,flag=0;
	int x,y;
	graph G;
	int s[1000];
	
	cin >> G.v >> n >> G.a ;

	for(int i=0; i<n; i++)
	{
		cin >> s[i];
	}
	for(int i=0; i<G.v; i++)
	{
		for(int j=0; j<G.v; j++)
		{
			G.arcs[i][j] = 0;
		}
	}
	for(int i=0; i<G.a; i++)
	{
		cin >> x >> y;
		G.arcs[x][y] = 1;
		G.arcs[y][x] = 1;
	}
	
	cin >> now >> d;
	
	for(int i=0; i<n; i++){
		if(d == s[i]){
			flag = 1;
			break;
		}
	}
	if(G.arcs[now][d] == 1){
		if(flag == 1)
			cout << "The city " << d << " can arrive safely!" << endl;
		else 
		{
			cout << "The city " << d << " is not safe!" << endl;
			
		}
	}
	else
	{
		if(flag == 1)
		cout << "The city " << d << " can not arrive safely!" << endl;
		else cout << "The city " << d << " is not safe!" << endl;
	}
	return 0;
}

```

用邻接矩阵存下城市，然后直接查询目标城市和目前所在的城市之间是否有通路



## 7-2 那就别担心了

```c++
#include<stdio.h>
int vis[1000], path[1000],vec[1000][1000];
int n, m, cnt, a, b;
int DFS(int x) {
    vis[x] = 1;
    if(path[x]) return path[x];
    for(int i=1;i<=n;i++)
        if(vec[x][i])
            path[x] += DFS(i);
    return path[x];
}
int main(){
    int i;
    scanf("%d %d",&n,&m);
    for(i=1;i<=m;i++) {
        scanf("%d %d",&a,&b);
        vec[a][b]=1;
    }
    scanf("%d %d",&a,&b);
    path[b] = 1;
    cnt = DFS(a);
    int flag = 1;
    for(i=1;i<=n;i++) {
        if(vis[i] && !path[i]) {
            flag = 0;
            break;
        }
    }
    printf("%d ",cnt);
    if(!flag)
        printf("No\n");
    else
        printf("Yes\n");
    return 0;
}

```

用记忆化搜索，将访问过的点到终点的路径数记录下来，下一次从别的路径访问到该点时就无需继续向后访问了，直接加上路径数即可。



## 7-3 扫雷游戏

```c++
#include<iostream>
using namespace std; 

int flag,xp;
int n,m,k,l,t;
int a[25][25];
bool c[25][25];

void F(int x,int y)
{
	if(c[x][y] == 1) 
		return;
	c[x][y] = 1;
	if(a[x][y] == 0){
	if(x-1 > 0)
		F(x-1,y);
	if(x+1 <= n)
		F(x+1,y);
	if(y-1 > 0)
		F(x,y-1);
	if(y+1 <= m)
		F(x,y+1);
	if(x-1 > 0 && y-1 > 0)
		F(x-1,y-1);
	if(x-1 > 0 && y+1 <= m)
		F(x-1,y+1);
	if(x+1 <= n && y-1 > 0)
		F(x+1,y-1);
	if(x+1 <= n && y+1 <= m)
		F(x+1,y+1);
	}
}


void fion(int x,int y)
{
	if(x-1>0 && a[x-1][y]!=-1)
		a[x-1][y]++;
	if(x+1<=n && a[x+1][y]!=-1)
		a[x+1][y]++;
	if(y-1>0 && a[x][y-1]!=-1)
		a[x][y-1]++;
	if(y+1<=m && a[x][y+1]!=-1)
		a[x][y+1]++;
	if(x-1>0 && y-1>0 && a[x-1][y-1]!=-1)
		a[x-1][y-1]++;
	if(x-1>0 && y+1<=m && a[x-1][y+1]!=-1)
		a[x-1][y+1]++;

}

inline int read()
{
```

## 7-4 01迷宫

```c++
#include<iostream>
#include<cstring>
using namespace std;

int n,m,ans[100010];
int x,y,f[1100][1100];
char s[1010][1010];

void dfs(int r,int c,int z,int lll){
    if (r<0 || r>=n || c<0 || c>=n || f[r][c]!=-1 || s[r][c]-'0'!=z)
		return;
    f[r][c]=lll;
	ans[lll]++;
    dfs(r-1,c,!z,lll);
	dfs(r+1,c,!z,lll);
	dfs(r,c-1,!z,lll);
	dfs(r,c+1,!z,lll);
}
int main()
{
    cin >> n >> m;
    for (int i=0;i<n;i++)
    	scanf("%s",s[i]);
    memset(f,-1,sizeof(f));
    for (int i=0;i<m;i++)
    {
       cin >> x >> y;
	   x--;
	   y--;
        if (f[x][y]==-1)
			dfs(x,y,s[x][y]-'0',i);
		else 
			ans[i]=ans[f[x][y]];
    }
    for (int i=0;i<m;i++)
    	cout << ans[i] << endl;
    	
    	
    return 0;
}
```
